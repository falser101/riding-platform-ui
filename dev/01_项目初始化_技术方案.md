### 2.2 用户表 (users)

| 字段 | 类型 | 约束 | 说明 |
|------|------|------|------|
| id | string | PK, auto | 用户ID |
| phone | string | unique, not null | 手机号 |
| password | string | not null | 密码（加密存储） |
| nickname | string | - | 昵称 |
| avatar | string | - | 头像URL |
| role | string | default: 'user' | 角色: user/admin |
| createTime | string | not null | 创建时间 |

### 2.3 Token表 (tokens)

| 字段 | 类型 | 约束 | 说明 |
|------|------|------|------|
| id | string | PK, auto | Token记录ID |
| userId | string | FK -> users.id | 关联用户ID |
| token | string | unique | 访问令牌 |
| expireTime | string | not null | 过期时间 |

## 5. 核心实现

### 5.1 Admin - Axios 封装 (src/api/request.ts)

```typescript
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { ElMessage } from 'element-plus'
import { getToken, removeToken } from '@/utils/storage'
import router from '@/router'

const service: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 15000,
  headers: { 'Content-Type': 'application/json' }
})

// 请求拦截器
service.interceptors.request.use(
  (config) => {
    const token = getToken()
    if (token) {
      config.headers.Authorization = `Bearer ${token}`
    }
    return config
  },
  (error) => Promise.reject(error)
)

// 响应拦截器
service.interceptors.response.use(
  (response: AxiosResponse) => {
    const { code, message } = response.data
    if (code === 0) {
      return response.data
    }
    // Token 过期
    if (code === 401) {
      removeToken()
      router.push('/login')
      ElMessage.error('登录已过期，请重新登录')
    }
    ElMessage.error(message || '请求失败')
    return Promise.reject(new Error(message))
  },
  (error) => {
    const status = error.response?.status
    if (status === 401) {
      removeToken()
      router.push('/login')
    }
    ElMessage.error(error.message || '网络错误')
    return Promise.reject(error)
  }
)

export default service
```

### 5.2 Admin - 用户状态管理 (src/stores/modules/user.ts)

```typescript
import { defineStore } from 'pinia'
import { getToken, setToken, removeToken } from '@/utils/storage'
import { getUserInfo, login as loginApi } from '@/api/modules/auth'

interface UserState {
  token: string | null
  userInfo: Record<string, any> | null
}

export const useUserStore = defineStore('user', {
  state: (): UserState => ({
    token: getToken(),
    userInfo: null
  }),

  actions: {
    async login(phone: string, code: string) {
      const res = await loginApi({ phone, code, type: 'admin' })
      this.token = res.data.token
      this.userInfo = res.data.userInfo
      setToken(res.data.token)
    },

    async fetchUserInfo() {
      if (!this.token) return
      const res = await getUserInfo()
      this.userInfo = res.data
    },

    logout() {
      this.token = null
      this.userInfo = null
      removeToken()
    }
  }
})
```

### 5.3 H5 - 小程序模拟器 (src/utils/mini-program.ts)

```typescript
// 简化的微信小程序 API 模拟
const AppStack: string[] = [] // 页面栈

export const miniProgram = {
  // 导航到页面
  navigateTo(url: string) {
    AppStack.push(url)
    history.pushState({}, '', url)
  },

  // 返回
  navigateBack(delta = 1) {
    for (let i = 0; i < delta; i++) {
      AppStack.pop()
    }
    const prevUrl = AppStack[AppStack.length - 1] || '/'
    history.pushState({}, '', prevUrl)
  },

  // 替换当前页
  redirectTo(url: string) {
    AppStack.pop()
    AppStack.push(url)
    history.replaceState({}, '', url)
  },

  // 获取当前页面路径
  getCurrentPage() {
    return AppStack[AppStack.length - 1] || '/'
  },

  // Tab 切换
  switchTab(url: string) {
    AppStack.length = 0
    AppStack.push(url)
    history.pushState({}, '', url)
  }
}

// 挂载到全局
declare global {
  interface Window {
    wx: typeof miniProgram
  }
}

window.wx = miniProgram
```
